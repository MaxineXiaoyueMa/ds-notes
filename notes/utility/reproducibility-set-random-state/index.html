numpy random seed vs. sklean random_state¶

import numpy as np
from sklearn.model_selection import KFold

random_seed = 1

np.random.seed(random_seed)

X = range(10)

No random_state¶

cv = KFold(n_splits = 5, shuffle = True)
for tr_ix, te_ix in cv.split(X):
    print (tr_ix, te_ix)

[0 1 2 3 4 7 8 9] [5 6]
[0 1 2 3 5 6 7 9] [4 8]
[0 1 3 4 5 6 7 8] [2 9]
[1 2 4 5 6 7 8 9] [0 3]
[0 2 3 4 5 6 8 9] [1 7]

cv = KFold(n_splits = 5, shuffle = True)
for tr_ix, te_ix in cv.split(X):
    print (tr_ix, te_ix)

[0 2 3 4 5 6 8 9] [1 7]
[0 1 2 4 5 6 7 8] [3 9]
[1 2 3 4 6 7 8 9] [0 5]
[0 1 3 5 6 7 8 9] [2 4]
